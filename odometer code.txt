def nums(a,b):
    lst=[]
   
    for i in range(a,b+1):
        i=str(i)
        num_lst=list(i)
        sorte=sorted(num_lst)
        k=0
        for j in range (len(num_lst)-1):
            if(num_lst[j]==num_lst[1+j]):
                k+=1
        if(num_lst==sorte and  0 not in num_lst and k==0):
            lst.append(i)
    odo_nums="\n".join(lst)
    return odo_nums
print(nums(100,1000))
